US Superstore Data Platform

A comprehensive data visualization platform built with Web Components and Express.js, featuring real-time data processing of over 10,000 sales records with interactive customer data cards using web components.

Custom Web Components

Implemented reusable data cards using Web Components API for displaying customer information: 

class DataCard extends HTMLElement {
    constructor() {
      super();
      this.root = this.attachShadow({ mode: "closed" });
      let clone = template.content.cloneNode(true);
      this.root.append(clone);
      
      // Overlay logic implementation
      const overlay = this.root.querySelector("#overlay");
      const moreInfoBtn = this.root.querySelector("#moreInfoBtn");
      
      moreInfoBtn.addEventListener("click", () => {
        this.updateOverlayContent();
        overlay.classList.remove("d-none");
      });
    }
  
    static get observedAttributes() {
      return [
        "customername",
        "salesfigure",
        "quantityfigure",
        "profitfigure"
      ];
    }
  }

Express.js Server

Built a RESTful API server using Express.js to handle data requests and CSV parsing: 

const express = require('express');
  const fs = require('fs');
  const app = express();
  const parse = require("csv-parse").parse;
  
  async function readAllUSSuperStore() {
     return new Promise((resolve, reject) => {
        const rData = [];
        fs.createReadStream("./data/US_Superstore.csv")
           .pipe(parse({ delimiter: ',', from_line: 2 }))
           .on('data', function (csvrow) {
              let newRecord = {
                 customerName: csvrow[6],
                 sales: parseFloat(csvrow[17]),
                 quantity: parseInt(csvrow[18]),
                 profit: parseFloat(csvrow[20])
              };
              rData.push(newRecord);
           })
           .on('end', () => resolve(rData))
           .on('error', (err) => reject(err));
     });
  }

Data Fetching

Implemented async data fetching and display logic: 

async function displayFirst10() {
    try {
      const response = await fetch("http://localhost:8081/allussuperstore");
      const data = await response.json();
      
      const dataCardDiv = document.querySelector("#datacards");
      dataCardDiv.innerHTML = "";
  
      data.slice(0, 10).forEach((record) => {
        const newDataCard = document.createElement("data-card");
        newDataCard.setAttribute("customername", record.customerName);
        newDataCard.setAttribute("salesfigure", record.sales.toFixed(2));
        dataCardDiv.append(newDataCard);
      });
    } catch (error) {
      console.error("Error fetching data:", error);
    }
  }
